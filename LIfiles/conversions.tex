\addsec{Conversions}

1 Tbsp = 3 tsp = 15 mL

\noindent 4 Tbsp = \quart C

\noindent 1 C = 8 fl oz

\noindent 1 gal = 4 qt = 8 pt = 16 C = 128 fl oz

\addsec{Substitutions}
{\bf Baking powder (1 tsp)}\begin{itemize}\itemindent10pt \itemsep0pt
\item\quart tsp baking soda, \half tsp cream of tartar, and \quart tsp cornstarch
\item\quart tsp baking soda and \half C buttermilk (replacing \half C non-acidic liquid)
\end{itemize}

\noindent{\bf Buttermilk (1 C)}\begin{itemize}\itemindent10pt \itemsep0pt
\item 1 C yogurt
\item 1 Tbsp lemon juice or vinegar and milk to make 1 C
\end{itemize}

\noindent{\bf Chocolate, baking or unsweetened (1 oz)}\begin{itemize}\itemindent10pt \itemsep0pt
\item 3 Tbsp cocoa and 1 Tbsp butter or vegetable oil
\end{itemize}

\noindent{\bf Coffee, strong (\quart C)}\begin{itemize}\itemindent10pt \itemsep0pt
\item 2 Tbsp instant espresso powder dissolved in 3 Tbsp hot water
\end{itemize}


\noindent{\bf Sour cream (1 C)}\begin{itemize}\itemindent10pt \itemsep0pt
\item 1 C plain yogurt
\item \tquart C buttermilk and \third C melted butter
\end{itemize}

\noindent{\bf Sugar, brown (1 C)}\begin{itemize}\itemindent10pt \itemsep0pt
\item1 C white sugar and \quart C molasses
\end{itemize}
